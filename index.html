Table 7.a Semantic Sparse KV Cache Sample Code (Dynamic Sparsity Part) 
def _create_causal_mask_for_probe_tokens(
    self,
    probe_positions: torch.Tensor,
    seq_length: int,
    *,
    dtype: torch.dtype,
    device: torch.device,
) -> torch.Tensor:
    mask = torch.full(
        (probe_positions.numel(), seq_length),
        torch.finfo(dtype).min,
        dtype=dtype,
        device=device,
    )
    token_positions = torch.arange(seq_length, device=device).unsqueeze(0)
    mask[token_positions <= probe_positions.unsqueeze(1)] = 0
    return mask

def _sample_sparse_probe_indices(self, segment_idx: int, valid_tokens: int, device: torch.device) -> torch.Tensor:
    recent_count = min(self.hier_kv_probe_recent, valid_tokens)
    recent_start = max(0, valid_tokens - recent_count)
    recent_ids = torch.arange(recent_start, valid_tokens, device=device, dtype=torch.long)

    random_pool = recent_start
    random_count = min(self.hier_kv_probe_random, random_pool)
    if random_count <= 0:
        return recent_ids

    generator = torch.Generator(device="cpu")
    generator.manual_seed((segment_idx + 1) * 1000003 + valid_tokens)
    random_ids = torch.randperm(random_pool, generator=generator, dtype=torch.long)[:random_count].to(device)
    return torch.unique(torch.cat([recent_ids, random_ids], dim=0), sorted=True)

def _select_sparse_hier_kv_token_indices(self, segment_idx: int, valid_tokens: int) -> torch.Tensor:
    if self.kv_cache1 is None or valid_tokens <= 1:
        return torch.arange(valid_tokens, device=self.kv_cache1[0]["k"].device if self.kv_cache1 is not None else "cpu")

    device = self.kv_cache1[0]["k"].device
    probe_ids = self._sample_sparse_probe_indices(segment_idx, valid_tokens, device=device)
    if probe_ids.numel() == 0:
        return torch.arange(valid_tokens, device=device)

    aggregated_scores = None
    score_blocks = 0
    head_dim = self.kv_cache1[0]["k"].shape[-1]
    scale = float(head_dim) ** -0.5
    score_norm = torch.flip(
        torch.arange(1, valid_tokens + 1, device=device, dtype=torch.float32),
        dims=(0,),
    )

    for cache_block in self.kv_cache1:
        key_states = cache_block["k"][:, :valid_tokens].detach()
        if key_states.numel() == 0:
            continue

        probe_query_states = key_states[:, probe_ids]
        probe_attn_weights = torch.einsum(
            "bphd,bthd->bhpt",
            probe_query_states.float(),
            key_states.float(),
        ) * scale
        probe_attn_weights = probe_attn_weights + self._create_causal_mask_for_probe_tokens(
            probe_ids,
            valid_tokens,
            dtype=probe_attn_weights.dtype,
            device=probe_attn_weights.device,
        ).unsqueeze(0).unsqueeze(0)
        probe_attn_weights = F.softmax(probe_attn_weights, dim=-1, dtype=torch.float32)

        token_attn_sum = probe_attn_weights.sum(dim=(1, 2))
        token_attn_mean = token_attn_sum / score_norm.unsqueeze(0)
        block_scores = token_attn_mean.mean(dim=0)
        aggregated_scores = block_scores if aggregated_scores is None else aggregated_scores + block_scores
        score_blocks += 1

    if aggregated_scores is None or score_blocks == 0:
        return torch.arange(valid_tokens, device=device)

    aggregated_scores = aggregated_scores / float(score_blocks)
    total_score = aggregated_scores.sum()
    if not torch.isfinite(total_score) or total_score.item() <= 0:
        return torch.arange(valid_tokens, device=device)

    sorted_scores, sorted_indices = torch.sort(aggregated_scores, descending=True)
    cumulative_scores = sorted_scores.cumsum(0)
    threshold = self.hier_kv_tau * total_score
    keep_count = int(torch.searchsorted(cumulative_scores, threshold).item()) + 1
    keep_count = max(1, min(valid_tokens, keep_count))
    return torch.sort(sorted_indices[:keep_count]).values
Table 7.b Semantic Sparse KV Cache Sample Code (semantic top-l retrieval Part) 
def _retrieve_hier_kv_chunks(
    self,
    conditional_dict: dict,
    *,
    exclude_segment_indices: Optional[set[int]] = None,
) -> List[int]:
    if len(self.hier_kv_chunks_cpu) == 0:
        return []
    query = self._extract_prompt_index_embedding(conditional_dict).detach()
    bank = torch.stack(self.hier_kv_prompt_index_gpu, dim=0)
    query = query.to(device=bank.device, dtype=bank.dtype)
    cosine_scores = bank @ query

    candidate_indices = []
    for idx, chunk in enumerate(self.hier_kv_chunks_cpu):
        if exclude_segment_indices is not None and int(chunk.get("segment_idx", -1)) in exclude_segment_indices:
            continue
        candidate_indices.append(idx)
    if len(candidate_indices) == 0:
        return []

    candidate_tensor = torch.tensor(candidate_indices, device=cosine_scores.device, dtype=torch.long)
    candidate_scores = cosine_scores.index_select(0, candidate_tensor)
    top_l = max(0, min(self.hier_kv_top_l, candidate_scores.numel()))
    if top_l == 0:
        return []

    top_scores, top_positions = torch.topk(candidate_scores, k=top_l, largest=True, sorted=True)
    selected_indices = candidate_tensor.index_select(0, top_positions).tolist()
    score_parts = [
        f"{chunk_idx}:{float(score.item()):.4f}"
        for chunk_idx, score in zip(selected_indices, top_scores)
    ]
    self._hkv_log(f"retrieved chunk_ids={selected_indices}, cosine=[{', '.join(score_parts)}]")
    return selected_indices

def _store_hier_kv_chunk(self, segment_idx: int, conditional_dict: dict):
    if self.kv_cache1 is None:
        self._hkv_log(f"skip store: kv_cache1 is None (segment={segment_idx})")
        return
    valid_tokens = int(self.kv_cache1[0]["local_end_index"].item())
    if valid_tokens <= 0:
        self._hkv_log(f"skip store: valid_tokens={valid_tokens} (segment={segment_idx})")
        return

    keep_indices = self._select_sparse_hier_kv_token_indices(segment_idx, valid_tokens)
    sparse_tokens = int(keep_indices.numel())
    if sparse_tokens <= 0:
        self._hkv_log(f"skip store: sparse_tokens={sparse_tokens} (segment={segment_idx})")
        return

    chunk = {
        "segment_idx": int(segment_idx),
        "valid_tokens": sparse_tokens,
        "source_tokens": valid_tokens,
        "k": [],
        "v": [],
    }
    for cache_block in self.kv_cache1:
        chunk["k"].append(cache_block["k"][:, keep_indices].detach().to(device="cpu", copy=True))
        chunk["v"].append(cache_block["v"][:, keep_indices].detach().to(device="cpu", copy=True))

    prompt_index = self._extract_prompt_index_embedding(conditional_dict).detach()
    self.hier_kv_chunks_cpu.append(chunk)
    self.hier_kv_prompt_index_gpu.append(prompt_index)

    if len(self.hier_kv_chunks_cpu) > self.hier_kv_max_chunks:
        self.hier_kv_chunks_cpu.pop(0)
        self.hier_kv_prompt_index_gpu.pop(0)

    self._hkv_log(
        f"stored chunk_id={len(self.hier_kv_chunks_cpu)-1}, "
        f"segment={segment_idx}, source_tokens={valid_tokens}, sparse_tokens={sparse_tokens}, location=cpu"
    )
Table 7.d Block Forcing Sample Code 
def _build_block_forcing_reference(
    self,
    conditional_dict: Dict[str, Any],
    target_num_frames: int,
) -> Tuple[Optional[torch.Tensor], Dict[str, Any]]:
    history = self.state.get("block_history", [])
    if not self.block_forcing_enabled or len(history) == 0:
        return None, {
            "block_forcing_history_size": len(history),
            "block_forcing_selected_chunk_starts": [],
            "block_forcing_selected_scores": [],
        }

    query = self._extract_prompt_index_embedding(conditional_dict).to(self.device)
    bank = torch.stack(
        [entry["prompt_index"].to(device=self.device, dtype=query.dtype) for entry in history],
        dim=0,
    )
    cosine_scores = bank @ query
    top_l = min(max(0, self.block_forcing_top_l), cosine_scores.numel())
    if top_l == 0:
        return None, {
            "block_forcing_history_size": len(history),
            "block_forcing_selected_chunk_starts": [],
            "block_forcing_selected_scores": [],
        }

    top_scores, top_positions = torch.topk(cosine_scores, k=top_l, largest=True, sorted=True)
    aligned_chunks: List[torch.Tensor] = []
    selected_chunk_starts: List[int] = []
    selected_scores: List[float] = []
    for score, history_idx in zip(top_scores.tolist(), top_positions.tolist()):
        history_entry = history[history_idx]
        history_chunk = history_entry["chunk"].to(device=self.device, dtype=self.dtype)
        aligned_chunks.append(self._temporal_align_block_chunk(history_chunk, target_num_frames))
        selected_chunk_starts.append(int(history_entry["chunk_start_frame"]))
        selected_scores.append(float(score))

    if len(aligned_chunks) == 0:
        return None, {
            "block_forcing_history_size": len(history),
            "block_forcing_selected_chunk_starts": [],
            "block_forcing_selected_scores": [],
        }

    x_cond = torch.stack(aligned_chunks, dim=0).mean(dim=0)
    return x_cond, {
        "block_forcing_history_size": len(history),
        "block_forcing_selected_chunk_starts": selected_chunk_starts,
        "block_forcing_selected_scores": selected_scores,
    }

def _compute_block_forcing_loss(
    self,
    flow_pred: torch.Tensor,
    noise: torch.Tensor,
    x_cond: torch.Tensor,
    gradient_mask: Optional[torch.Tensor] = None,
) -> torch.Tensor:
    target_flow = noise - self.block_forcing_gamma * x_cond.to(device=noise.device, dtype=noise.dtype)
    err = (flow_pred - target_flow) ** 2
    if gradient_mask is not None:
        return err[gradient_mask].mean()
    return err.mean()

def _get_block_context_noise_timestep(self, block_index: int, num_blocks: int) -> int:
    if not self.noise_schedule_enabled or num_blocks <= 1:
        return int(self.context_noise)

    progress = float(block_index) / float(max(1, num_blocks - 1))
    if self.noise_schedule_type == "linear":
        scheduled = self.noise_schedule_eps_min + (self.noise_schedule_eps_max - self.noise_schedule_eps_min) * progress
    else:
        scheduled = self.noise_schedule_eps_min + 0.5 * (self.noise_schedule_eps_max - self.noise_schedule_eps_min) * (
            1.0 - torch.cos(torch.tensor(progress * torch.pi)).item()
        )
    return int(round(scheduled))

def _apply_boundary_noise_shuffle(self, noise: torch.Tensor) -> torch.Tensor:
    if not self.noise_shuffle_enabled or self.num_frame_per_block <= 1:
        return noise

    shuffled_noise = noise.clone()
    total_frames = shuffled_noise.shape[1]
    num_blocks = total_frames // self.num_frame_per_block
    if num_blocks <= 1:
        return shuffled_noise

    window_size = min(self.noise_shuffle_window_size, self.num_frame_per_block)
    if window_size <= 1:
        return shuffled_noise

    generator = torch.Generator(device="cpu")
    for boundary_idx in range(num_blocks - 1):
        current_block_start = boundary_idx * self.num_frame_per_block
        next_block_start = (boundary_idx + 1) * self.num_frame_per_block
        current_window = slice(next_block_start - window_size, next_block_start)
        next_window = slice(next_block_start, next_block_start + window_size)

        generator.manual_seed(self.noise_shuffle_seed + boundary_idx * 2)
        current_perm = torch.randperm(window_size, generator=generator)
        generator.manual_seed(self.noise_shuffle_seed + boundary_idx * 2 + 1)
        next_perm = torch.randperm(window_size, generator=generator)

        shuffled_noise[:, current_window] = shuffled_noise[:, current_window][:, current_perm.to(device=shuffled_noise.device)]
        shuffled_noise[:, next_window] = shuffled_noise[:, next_window][:, next_perm.to(device=shuffled_noise.device)]

    if DEBUG and (not dist.is_initialized() or dist.get_rank() == 0):
        print(
            f"[NoiseShuffle] applied boundary shuffle with window_size={window_size}, "
            f"num_blocks={num_blocks}, seed={self.noise_shuffle_seed}"
        )
    return shuffled_noise
